99  FT h = mOptions.mHInit;
100  const snowgoose::Box<double>& box = *(mProblem.mBox);
101 
102  // One step
103  auto step = [&] () {
104  FT xd = 0;
105  for (int i = 0; i < n; i++) {
106  FT y = x[i] - h;
107  if (y < box.mA[i]) {
108  y = box.mA[i];
109  }
110  FT tmp = x[i];
111  x[i] = y;
112  FT fn = obj->func(x);
113  if (fn >= fcur) {
114  x[i] = tmp;
115  } else {
116  fcur = fn;
117  xd += h * h;
118  continue;
119  }
120 
121  y = x[i] + h;
122  if (y > box.mB[i]) {
123  y = box.mB[i];
124  }
125  tmp = x[i];
126  x[i] = y;
127  fn = obj->func(x);
128  if (fn >= fcur) {

131  fcur = fn;

133  continue;
134  }
